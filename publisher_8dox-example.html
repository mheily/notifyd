#include <state.h>
int main(int argc, char *argv[])
{
const char *name = "user.example.end_of_the_world_as_we_know_it";
const char *state = "I feel fine";
state_init(0, 0);
state_bind(name);
state_publish(name, state, strlen(state));
}
